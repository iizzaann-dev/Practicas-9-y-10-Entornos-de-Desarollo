package calculadora;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class RectanguloTest {

    // positivos
    @Test
    void testPositivos() {
        Rectangulo r = new Rectangulo(4, 5);
        assertEquals(20, r.area());
        assertEquals(18, r.perimetro());
    }

    // negativos
    @Test
    void testNegativos() {
        Rectangulo r = new Rectangulo(-4, 5);
        assertEquals(-1, r.area());
        assertEquals(-1, r.perimetro());
    }

    // ceros
    @Test
    void testCeros() {
        Rectangulo r = new Rectangulo(0, 5);
        assertEquals(0, r.area());
        assertEquals(0, r.perimetro());
    }

    // aserciones erróneas (intencional)
    @Test
    void testFallosIntencionales() {
        Rectangulo r = new Rectangulo(4, 5);

        assertEquals(999, r.area());
        assertEquals(999, r.perimetro());
    }
}
